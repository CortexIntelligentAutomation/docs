{{- /* Creates a hyperlink element with the provided link. This is currently not versioned. */}} {{- $tomlPath := .Get "path" }} {{- $url := urls.Parse (index .Site.Data.urls (split ($tomlPath) ".")) }} {{- $link := (printf "%s://%s%s" $url.Scheme $url.Host $url.Path) }} {{- /* Get the current version, and whether the website is currently nested (has /docs prepended). */}} {{- $isNested := false }} {{- $currentVer := index (findRE `^\/docs\/([0-9\.]*|latest)\/` .Page.RelPermalink 1) 0 }} {{- if (not $currentVer) }} {{- $currentVer = index (findRE `^\/docs\/docs\/([0-9\.]*|latest)\/` .Page.RelPermalink 1) 0 }} {{- $currentVer = replaceRE `^\/docs` "" $currentVer 1 }} {{- if (or $currentVer (index (findRE `^\/docs\/[^\/]*\/` .Page.RelPermalink 1) 0)) }} {{- $isNested = true }} {{- end }} {{- end }} {{- /* Overwrites the current version if the version parameter is provided, or latest if no current version. */}} {{- $version := .Get "version" }} {{- if $version }} {{- range where .Site.Params.versions "version" $version }} {{- $currentVer = .url }} {{- end }} {{- else if (not $currentVer) }} {{- range where .Site.Params.versions "latest" true }} {{- $currentVer = .url }} {{- end }} {{- end }} {{- /* Set the url path, which is used to check whether the file exists. */}} {{- $urlPath := replaceRE `^/docs/` $currentVer $url.Path 1 }} {{- /* Sets the file paths where the actual file could be, which will be used later to check if the file exist. */}} {{- $mdFilePath := printf "%s%s" (replaceRE "/$" "" (printf "%s%s" $url.Host $urlPath)) ".md" }} {{- $indexFilePath := printf "%s%s" (replaceRE "/$" "" (printf "%s%s" $url.Host $urlPath)) "/_index.md" }} {{- $page := .Site.GetPage $mdFilePath }} {{- $pageIndex := .Site.GetPage $indexFilePath }} {{- /* add the anchor if the url contains fragments. */}} {{- if ne $url.Fragment "" }} {{- $urlPath = (printf "%s#%s" $urlPath $url.Fragment) }} {{- end }} {{- /* Check if the URL does not exist */}} {{- if eq $link (printf "%s://" $url.Scheme) }} {{- errorf "ahref.html: No Url found in urls.toml matching with %q. Stacktrace: %s" $tomlPath . }} {{- /* Check if it is an external URL to microsoft docs. */}} {{- else if hasPrefix $url "https://learn.microsoft.com/en-us/dotnet/api/" }} {{- /* Make the link for the current version */}} {{- range where .Site.Params.versions "url" $currentVer }} {{- $link = (printf "%s?view=%s" $link .dotNetVersion) }} {{- if ne $url.Fragment "" }} {{- $link = (printf "%s#%s" $link $url.Fragment) }} {{- end }} {{- $link }} {{- end }} {{ .Get "title" }} {{- /* Check if it is an internal docs URL. */}} {{- else if hasPrefix $mdFilePath "/docs/" }} {{- if (not (or ($page) ($pageIndex))) }} {{- errorf "ahref.html: Missing md file for url %q found in urls.toml. One of the following file must exist for the link to work: %q or %q." $mdFilePath $mdFilePath $indexFilePath }} {{- end }} {{- if $isNested }} {{- $urlPath = printf "%s%s" "/docs" $urlPath }} {{- end }} {{ .Get "title" }} {{- /* If it is a generic external url. */}} {{- else }} {{ .Get "title" }} {{- end }}
